// example.
#[allow(dead_code)]
fn bad_add(a: i32, b: i32) -> i32 {
    a - b
}

    // Note this useful idiom: importing names from outer (for mod tests) scope.
    use super::*;

    #[test]

    }

    #[test]
    fn test_bad_add() {
